# Base class
class PalindromeChecker:

    def __init__(self, value):
        self.value = value

    def is_palindrome(self):
        raise NotImplementedError("Subclass must implement this method")


# Subclass for string palindrome check
class StringPalindrome(PalindromeChecker):

    def is_palindrome(self):
        text = str(self.value).lower().replace(" ", "")
        return text == text[::-1]


# Subclass for integer palindrome check
class IntegerPalindrome(PalindromeChecker):

    def is_palindrome(self):
        num_str = str(self.value)
        return num_str == num_str[::-1]


# --- Main Program ---
user_input = input("Enter a string or number: ")

if user_input.isdigit():
    checker = IntegerPalindrome(int(user_input))
else:
    checker = StringPalindrome(user_input)

if checker.is_palindrome():
    print(f"{user_input} is a palindrome ‚úî")
else:
    print(f"{user_input} is not a palindrome ‚úñ")
